// 导入Excel功能 async function startImport() { const fileInput = document.getElementById('import-file'); const passwordInput = document.getElementById('import-password'); const userSelect = document.getElementById('import-user'); if (!fileInput.files.length) { alert('请选择一个Excel文件'); return; } const file = fileInput.files[0]; const password = passwordInput.value || ''; const userId = userSelect.value; if (!userId) { alert('请选择导入到的用户'); return; } const progressDiv = document.getElementById('import-progress'); const messageDiv = document.getElementById('import-message'); const resultDiv = document.getElementById('import-result'); progressDiv.style.display = 'block'; messageDiv.textContent = '📤 正在上传文件...'; resultDiv.textContent = ''; try { const result = await importExcelAPI(file, password, userId); if (result.error) { messageDiv.textContent = '❌ 导入失败: ' + result.error; messageDiv.style.color = '#dc2626'; } else { messageDiv.textContent = `✅ ${result.message}`; messageDiv.style.color = '#059669'; // 显示导入统计 let resultText = `✓ 成功导入: ${result.imported} 条记录`; if (result.errors && result.errors.length > 0) { resultText += `\n\n⚠️ 警告 (${result.errors.length}):\n`; passwordInput.value = ''; userSelect.value = ''; // 刷新数据 loadHomeData(); }, 2000); } } catch (error) { messageDiv.textContent = '❌ 错误: ' + error.message; messageDiv.style.color = '#dc2626'; resultDiv.textContent = error.toString();
